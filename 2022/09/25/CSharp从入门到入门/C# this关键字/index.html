C# this关键字

在 C# 中，可以使用 this 关键字来表示当前对象，日常开发中我们可以使用 this 关键字来访问类中的成员属性以及函数。不仅如此 this 关键字还有一些其它的用法，下面就通过一些示例来分别演示一下。

1) 使用 this 表示当前类的对象

  8.         {
  9.             Website site = new Website("C语言中文网","http://c.biancheng.net/");
  10.             site.Display();
  11.         }
  12.     }
  13.     public class Website
  14.     {
  15.         private string name;
  16.         private string url;
  17.         public Website(string n, string u){
  18.             this.name = n;
  19.             this.url = u;
  20.         }
  21.         public void Display(){
  22.             Console.WriteLine(name +" "+ url);
  23.         }
  24.     }
  25. }
using System;

namespace c.biancheng.net
{
    class Demo
    {
        static void Main(string[] args) 
        {
            Website site = new Website("C语言中文网","http://c.biancheng.net/");
            site.Display();

2) 使用 this 关键字串联构造函数

复制纯文本复制
  1. using System;
  2.  
  3. namespace c.biancheng.net
  4. {
  5.     class Demo
  6.     {
  7.         static void Main(string[] args) 
  8.         {
  9.             Test test = new Test("C语言中文网");
  10.         }
  11.     }
  12.     public class Test
  13.     {
  14.         public Test()
  15.         {
  16.             Console.WriteLine("无参构造函数");
  17.         }
  18.         // 这里的 this()代表无参构造函数 Test()
  19. 　　      // 先执行 Test()，后执行 Test(string text)
  20.         public Test(string text) : this()
  21.         {
  22.             Console.WriteLine(text);
  23.             Console.WriteLine("实例构造函数");
  24.         }
  25.     }
  26. }
using System;

namespace c.biancheng.net
{
    class Demo
    {
        static void Main(string[] args) 
        {
            Test test = new Test("C语言中文网");
        }
    }
    public class Test
    {
        public Test()
        {
            Console.WriteLine("无参构造函数");
        }
        // 这里的 this()代表无参构造函数 Test()
　　      // 先执行 Test()，后执行 Test(string text)
        public Test(string text) : this()
        {
            Console.WriteLine(text);
            Console.WriteLine("实例构造函数");
        }
    }
}

3) 使用 this 关键字作为类的索引器

复制纯文本复制
  1. using System;
  2.  
  3. namespace c.biancheng.net
  4. {
  5.     class Demo
  6.     {
  7.         static void Main(string[] args) 
  8.         {
  9.             Test a = new Test();
  10.             Console.WriteLine("Temp0:{0}, Temp1:{1}", a[0], a[1]);
  11.             a[0] = 15;
  12.             a[1] = 20;
  13.             Console.WriteLine("Temp0:{0}, Temp1:{1}", a[0], a[1]);
  14.         }
  15.     }
  16.     public class Test
  17.     {
  18.         int Temp0;
  19.         int Temp1;
  20.         public int this[int index]
  21.         {
  22.             get
  23.             {
  24.                 return (0 == index) ? Temp0 : Temp1;
  25.             }
  26.     
  27.             set
  28.             {
  29.                 if (0==index)
  30.                     Temp0 = value;
  31.                 else
  32.                     Temp1 = value;
  33.             }
  34.         }
  35.     }
  36. }
using System;

namespace c.biancheng.net
{
    class Demo
    {
        static void Main(string[] args) 
        {
            Test a = new Test();
            Console.WriteLine("Temp0:{0}, Temp1:{1}", a[0], a[1]);
            a[0] = 15;
            a[1] = 20;
            Console.WriteLine("Temp0:{0}, Temp1:{1}", a[0], a[1]);
        }
    }
    public class Test
    {
        int Temp0;
        int Temp1;
        public int this[int index]
        {
            get
            {
                return (0 == index) ? Temp0 : Temp1;
            }
    
            set
            {
                if (0==index)
                    Temp0 = value;
                else
                    Temp1 = value;
            }
        }
    }
}
运行结果如下：

Temp0:0, Temp1:0
Temp0:15, Temp1:20

4) 使用 this 关键字作为原始类型的扩展方法

复制纯文本复制
  1. using System;
  2.  
  3. namespace c.biancheng.net
  4. {
  5.     class Demo
  6.     {
  7.         static void Main(string[] args) 
  8.         {
  9.             string str = "C语言中文网";
  10.             string newstr = str.ExpandString();
  11.             Console.WriteLine(newstr);
  12.         }
  13.     }
  14.     public static class Test
  15.     {
  16.         public static string ExpandString(this string name)
  17.         {
  18.             return name+" http://c.biancheng.net/";
  19.         }
  20.     }
  21. }
using System;

namespace c.biancheng.net
{
    class Demo
    {
        static void Main(string[] args) 
        {
            string str = "C语言中文网";
            string newstr = str.ExpandString();
            Console.WriteLine(newstr);
        }
    }
    public static class Test
    {
        public static string ExpandString(this string name)
        {
            return name+" http://c.biancheng.net/";
        }
    }
}
